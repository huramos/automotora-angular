import { Component, OnInit, ElementRef } from '@angular/core'; // Importa el decorador `Component`, ciclo de vida `OnInit` y manipulación directa del DOM con `ElementRef`.
import { CommonModule } from '@angular/common'; // Módulo común de Angular para directivas como `ngIf` y `ngFor`.
import { ReactiveFormsModule, FormBuilder, FormGroup, Validators } from '@angular/forms'; // Módulos para formularios reactivos.
import { RouterModule, Router } from '@angular/router'; // Manejo de rutas.

declare var bootstrap: any; // Declara Bootstrap como variable para manipulación directa.

@Component({
  selector: 'app-login-form', // Selector que identifica este componente en las plantillas HTML.
  standalone: true, // Define que este componente no necesita estar dentro de un módulo específico.
  imports: [CommonModule, RouterModule, ReactiveFormsModule], // Módulos necesarios para este componente.
  templateUrl: './login-form.component.html', // Archivo HTML asociado al componente.
  styleUrls: ['./login-form.component.css'] // Archivo CSS asociado al componente.
})
export class LoginFormComponent implements OnInit {
  loginForm!: FormGroup; // Formulario reactivo para manejar los campos de inicio de sesión.
  usuarios: any[] = []; // Lista de usuarios almacenados en localStorage.

  // Constructor para inicializar dependencias
  constructor(private fb: FormBuilder, private router: Router, private el: ElementRef) {
    // Carga los usuarios almacenados en localStorage si existen
    if (typeof localStorage !== 'undefined') {
      const usuariosGuardados = localStorage.getItem('usuarios');
      this.usuarios = usuariosGuardados ? JSON.parse(usuariosGuardados) : [];
    }
  }

  // Método del ciclo de vida `OnInit`
  ngOnInit() {
    // Inicializa el formulario con dos campos: email/usuario y contraseña
    this.loginForm = this.fb.group({
      emailOrUsername: ['', Validators.required], // Campo obligatorio para email o usuario
      password: ['', Validators.required] // Campo obligatorio para la contraseña
    });
  }

  // Método para manejar el inicio de sesión
  iniciarSesion(emailOrUsername: string, password: string): boolean {
    // Busca un usuario en localStorage que coincida con el email/usuario y la contraseña
    const usuario = this.usuarios.find(user =>
      (user.email === emailOrUsername || user.username === emailOrUsername) &&
      user.password === password
    );

    if (usuario) {
      // Si el usuario existe, muestra un mensaje de éxito y redirige a `/home`
      this.showToast('Inicio de sesión exitoso.', 'success');
      this.router.navigate(['/home']);
      return true;
    } else {
      // Si no existe, muestra un mensaje de error
      this.showToast('Email/Usuario o contraseña incorrectos.', 'danger');
      return false;
    }
  }

  // Método que se ejecuta al enviar el formulario
  onSubmit() {
    if (this.loginForm.valid) { // Verifica si el formulario es válido
      const emailOrUsernameControl = this.loginForm.get('emailOrUsername'); // Obtiene el valor del campo de email/usuario
      const passwordControl = this.loginForm.get('password'); // Obtiene el valor del campo de contraseña

      if (emailOrUsernameControl && passwordControl) {
        const emailOrUsername = emailOrUsernameControl.value;
        const password = passwordControl.value;

        const inicioExitoso = this.iniciarSesion(emailOrUsername, password); // Intenta iniciar sesión
        if (inicioExitoso) {
          this.loginForm.reset(); // Resetea el formulario en caso de éxito
        }
      } else {
        console.error('Form controls are missing'); // Muestra un error si los controles del formulario no están disponibles
        this.showToast('Formulario no válido.', 'danger');
      }
    } else {
      this.showToast('Formulario no válido.', 'danger'); // Muestra un mensaje de error si el formulario es inválido
    }
  }

  // Método para redirigir al formulario de registro
  navigateToRegister() {
    this.router.navigateByUrl('/register').then(() => {
      window.location.reload(); // Recarga la página después de redirigir
    });
  }

  // Método para mostrar un mensaje emergente (toast)
  showToast(message: string, type: 'success' | 'danger') {
    const toastContainer = this.el.nativeElement.querySelector('#toast-container'); // Selecciona el contenedor de los toasts
    const toastElement = document.createElement('div'); // Crea un elemento div para el toast
    toastElement.className = `toast align-items-center text-bg-${type} border-0`; // Aplica clases según el tipo de mensaje
    toastElement.role = 'alert';
    toastElement.ariaLive = 'assertive';
    toastElement.ariaAtomic = 'true';
    toastElement.innerHTML = `
      <div class="d-flex">
        <div class="toast-body">
          ${message} <!-- Mensaje del toast -->
        </div>
        <button type="button" class="btn-close btn-close-white me-2 m-auto" data-bs-dismiss="toast" aria-label="Close"></button>
      </div>
    `;
    toastContainer.appendChild(toastElement); // Añade el toast al contenedor

    const toastBootstrap = new bootstrap.Toast(toastElement); // Inicializa el toast usando Bootstrap
    toastBootstrap.show(); // Muestra el toast

    setTimeout(() => {
      toastBootstrap.hide(); // Oculta el toast después de 3 segundos
      toastElement.remove(); // Elimina el elemento del DOM
    }, 3000);
  }
}
